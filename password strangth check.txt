    </p>
    <h2 class="sec-title">Think like an attacker, defend like a pro</h2>
    <p class="sec-desc">Six short lessons that make the biggest difference to your account security.</p>
 
    <div class="tips-grid">
      <div class="gc tip-card">
        <div class="tip-icon"><svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="white" stroke-width="2.2" stroke-linecap="round"><path d="M10.29 3.86L1.82 18a2 2 0 0 0 1.71 3h16.94a2 2 0 0 0 1.71-3L13.71 3.86a2 2 0 0 0-3.42 0z"/><line x1="12" y1="9" x2="12" y2="13"/><line x1="12" y1="17" x2="12.01" y2="17"/></svg></div>
        <h3>Why strong passwords matter</h3>
        <p>A weak password can be cracked in seconds with modern hardware. Strong, unique passwords are the single biggest barrier between attackers and your accounts.</p>
      </div>
      <div class="gc tip-card">
        <div class="tip-icon"><svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="white" stroke-width="2.2" stroke-linecap="round"><path d="M8 3H5a2 2 0 0 0-2 2v3m18 0V5a2 2 0 0 0-2-2h-3m0 18h3a2 2 0 0 0 2-2v-3M3 16v3a2 2 0 0 0 2 2h3"/></svg></div>
        <h3>Common password mistakes</h3>
        <p>Reusing passwords, using personal info, sequential keys like "123456" or "qwerty", and simple substitutions ("p@ssw0rd") are the first things attackers try.</p>
      </div>
      <div class="gc tip-card">
        <div class="tip-icon"><svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="white" stroke-width="2.2" stroke-linecap="round"><polyline points="9 11 12 14 22 4"/><path d="M21 12v7a2 2 0 0 1-2 2H5a2 2 0 0 1-2-2V5a2 2 0 0 1 2-2h11"/></svg></div>
        <h3>Best practices</h3>
        <p>Use long random passphrases, mix character types, never reuse passwords across sites, and rotate credentials immediately after any breach notice.</p>
      </div>
      <div class="gc tip-card">
        <div class="tip-icon"><svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="white" stroke-width="2.2" stroke-linecap="round"><rect x="3" y="11" width="18" height="11" rx="2"/><path d="M7 11V7a5 5 0 0 1 10 0v4"/></svg></div>
        <h3>Benefits of password managers</h3>
        <p>A password manager generates and stores unique, high-entropy passwords for every account, so you never have to remember or reuse a single one.</p>
      </div>
      <div class="gc tip-card">
        <div class="tip-icon"><svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="white" stroke-width="2.2" stroke-linecap="round"><rect x="5" y="2" width="14" height="20" rx="2" ry="2"/><line x1="12" y1="18" x2="12.01" y2="18"/></svg></div>
        <h3>Two-Factor Authentication (2FA)</h3>
        <p>Even a strong password can leak. 2FA adds a second proof of identity — an app code or hardware key — that stops most account takeovers cold.</p>
      </div>
      <div class="gc tip-card">
        <div class="tip-icon"><svg width="20" height="20" viewBox="0 0 24 24" fill="none" stroke="white" stroke-width="2.2" stroke-linecap="round"><polyline points="23 4 23 10 17 10"/><polyline points="1 20 1 14 7 14"/><path d="M3.51 9a9 9 0 0 1 14.85-3.36L23 10M1 14l4.64 4.36A9 9 0 0 0 20.49 15"/></svg></div>
        <h3>Update regularly</h3>
        <p>Audit and rotate critical passwords periodically, especially for email, banking, and admin accounts — your weakest link defines your real security.</p>
      </div>
    </div>

/* ── Analysis ── */
function hasSeq(pw){
  const lo=pw.toLowerCase();
  for(const s of SEQS){for(let i=0;i<=s.length-4;i++){const c=s.slice(i,i+4);if(lo.includes(c)||lo.includes(c.split("").reverse().join("")))return true;}}
  for(let i=0;i<pw.length-3;i++){const a=pw.charCodeAt(i),b=pw.charCodeAt(i+1),c=pw.charCodeAt(i+2),d=pw.charCodeAt(i+3);if(b===a+1&&c===b+1&&d===c+1)return true;if(b===a-1&&c===b-1&&d===c-1)return true;}
  return false;
}
function hasRep(pw){return/(.)\1{2,}/.test(pw)}
function dictWord(pw){const lo=pw.toLowerCase();return DICT.find(w=>w.length>=4&&lo.includes(w))||null}
function isCommon(pw){return COMMON.has(pw.toLowerCase())}
function entropy(pw){if(!pw)return 0;let p=0;if(/[a-z]/.test(pw))p+=26;if(/[A-Z]/.test(pw))p+=26;if(/[0-9]/.test(pw))p+=10;if(/[^a-zA-Z0-9]/.test(pw))p+=32;return p?pw.length*Math.log2(p):0}
function crackTime(bits){
  if(bits<=0)return"Instantly";
  const s=Math.pow(2,bits)/1e10/2;if(s<1)return"Instantly";
  const U=[{l:"second",s:1},{l:"minute",s:60},{l:"hour",s:3600},{l:"day",s:86400},{l:"month",s:2592000},{l:"year",s:31536000},{l:"century",s:3153600000}];
  if(s>3.15e12)return"Trillions of years";
  let u=U[0];for(const x of U){if(s>=x.s)u=x;}
  const v=s/u.s,r=v>=100?Math.round(v):Math.round(v*10)/10;
  return`${r} ${u.l}${r===1?"":"s"}`;
}
function rules(pw){return{length:pw.length>=8,upper:/[A-Z]/.test(pw),lower:/[a-z]/.test(pw),number:/[0-9]/.test(pw),special:/[^a-zA-Z0-9]/.test(pw)}}
 
function analyze(pw){
  const r=rules(pw), passed=Object.values(r).filter(Boolean).length;
  const ent=entropy(pw);
  const warns=[];
  if(pw&&isCommon(pw))warns.push({ic:"warn",t:"This is one of the most commonly used passwords in the world."});
  if(pw&&hasRep(pw))warns.push({ic:"repeat",t:'Contains repeated characters (e.g. "aaa"), which weakens unpredictability.'});
  if(pw&&hasSeq(pw))warns.push({ic:"trend",t:'Contains a sequential pattern (e.g. "1234" or "qwerty").'});
  const dw=pw?dictWord(pw):null;
  if(dw)warns.push({ic:"book",t:`Contains the dictionary word "${dw}", which is easy to guess.`});
 
  let score=0;
  if(pw){
    score+=Math.min(40,(ent/80)*40);
    score+=passed*8;
    score+=Math.min(20,pw.length*1.2);
    if(isCommon(pw))score-=50;
    if(hasSeq(pw))score-=15;
    if(hasRep(pw))score-=10;
    if(dw)score-=12;
    score=Math.max(0,Math.min(100,Math.round(score)));
  }
  let lv,lb,em;
  if(!pw){lv=0;lb="No password";em="—";}
  else if(score<35){lv=1;lb="Weak";em="🔴";}
  else if(score<60){lv=2;lb="Fair";em="🟠";}
  else if(score<80){lv=3;lb="Good";em="🟡";}
  else{lv=4;lb="Strong";em="🟢";}
 
  const sugs=[];
  if(pw){
    if(!r.length)sugs.push("Make it at least 8 characters long.");
    if(!r.upper)sugs.push("Add at least one uppercase letter (A–Z).");
    if(!r.lower)sugs.push("Add at least one lowercase letter (a–z).");
    if(!r.number)sugs.push("Add at least one number (0–9).");
    if(!r.special)sugs.push("Add a special character like ! @ # $ % ^ &.");
    if(pw.length<12&&passed===5)sugs.push("Extend to 12+ characters for an even higher score.");
    if(isCommon(pw))sugs.push("Avoid this password — it appears in known leaked password lists.");
    if(hasSeq(pw))sugs.push("Break up sequential patterns like \"1234\" or \"qwerty\".");
    if(hasRep(pw))sugs.push("Avoid repeating the same character multiple times.");
    if(dw)sugs.push("Replace or break up the dictionary word with random characters.");
    if(!sugs.length)sugs.push("Excellent! Use a password manager to keep it unique per site.");
  }
  return{rules:r,passed,entropy:ent,score,lv,lb,em,crack:crackTime(ent),warns,sugs};
}

function initParticles(){
  const cv=document.getElementById("bg-canvas"),ctx=cv.getContext("2d");
  const C=["#ec4899","#a855f7","#22d3ee"];
  let parts=[];
  const nomotion=window.matchMedia("(prefers-reduced-motion:reduce)").matches;
  function resize(){
    cv.width=window.innerWidth;cv.height=window.innerHeight;
    const n=Math.min(80,Math.floor(cv.width*cv.height/20000));
    parts=Array.from({length:n},()=>({x:Math.random()*cv.width,y:Math.random()*cv.height,r:Math.random()*1.8+.5,c:C[Math.floor(Math.random()*3)],vx:(Math.random()-.5)*.22,vy:(Math.random()-.5)*.22,a:Math.random()*.5+.2}));
  }
  function step(){
    ctx.clearRect(0,0,cv.width,cv.height);
    for(const p of parts){
      p.x+=p.vx;p.y+=p.vy;
      if(p.x<0)p.x=cv.width;if(p.x>cv.width)p.x=0;
      if(p.y<0)p.y=cv.height;if(p.y>cv.height)p.y=0;
      ctx.beginPath();ctx.arc(p.x,p.y,p.r,0,Math.PI*2);
      ctx.fillStyle=p.c;ctx.globalAlpha=p.a;ctx.fill();
    }
    ctx.globalAlpha=1;
    if(!nomotion)requestAnimationFrame(step);
  }
  resize();window.addEventListener("resize",resize);
  if(!nomotion)step();
}
